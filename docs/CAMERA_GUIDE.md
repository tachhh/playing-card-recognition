### 2. ‡∏™‡πà‡∏á‡∏†‡∏≤‡∏û‡πÉ‡∏´‡πâ AI ‡∏ó‡∏≥‡∏ô‡∏≤‡∏¢

```python
def predict_card(image_rgb):
    # 1. ‡πÅ‡∏õ‡∏•‡∏á‡πÄ‡∏õ‡πá‡∏ô PIL Image
    pil_image = Image.fromarray(image_rgb)
    
    # 2. Transform ‡∏†‡∏≤‡∏û
    input_tensor = transform(pil_image).unsqueeze(0)
    # unsqueeze(0) = ‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏°‡∏¥‡∏ï‡∏¥ Batch (AI ‡∏£‡∏±‡∏ö‡πÑ‡∏î‡πâ‡∏´‡∏•‡∏≤‡∏¢‡∏£‡∏π‡∏õ‡∏û‡∏£‡πâ‡∏≠‡∏°‡∏Å‡∏±‡∏ô)
    
    # 3. ‡∏™‡πà‡∏á‡πÄ‡∏Ç‡πâ‡∏≤ GPU/CPU
    input_tensor = input_tensor.to(device)
    
    # 4. ‡∏ó‡∏≥‡∏ô‡∏≤‡∏¢ (‡πÑ‡∏°‡πà‡∏Ñ‡∏≥‡∏ô‡∏ß‡∏ì gradient ‡πÄ‡∏û‡∏£‡∏≤‡∏∞‡πÑ‡∏°‡πà‡πÑ‡∏î‡πâ‡∏ù‡∏∂‡∏Å)
    with torch.no_grad():
        outputs = model(input_tensor)  # ‡πÑ‡∏î‡πâ‡∏ï‡∏±‡∏ß‡πÄ‡∏•‡∏Ç 53 ‡∏ï‡∏±‡∏ß (53 ‡∏ä‡∏ô‡∏¥‡∏î‡πÑ‡∏°‡πâ)

        probabilities = torch.nn.functional.softmax(outputs, dim=1)
```
